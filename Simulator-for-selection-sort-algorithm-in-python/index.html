Sorting list using Selection Sort in Python
Initialize
Enter the size of list
Step Execution
A = list(map(int,input().split()))
for i in range(len(A)):
min_idx = i
for j in range(i+1, len(A)):
if A[min_idx] > A[j]:
min_idx = j
A[i], A[min_idx] = A[min_idx], A[i]
print ("Array is sorted")
